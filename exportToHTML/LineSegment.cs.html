1    using System;
2    using UnityEngine;
3    
4    namespace Shapes
5    {
6        [Serializable]
7        public class LineSegment
8        {
9            public Vector2 A;
10           public Vector2 B;
11   
12           public LineSegment(Vector2 a, Vector2 b)
13           {
14               A = a;
15               B = b;
16           }
17   
18           public static bool ArePointsOnSegment(Vector2 p, Vector2 q, Vector2 r)
19           {
20               return q.x <= Mathf.Max(p.x, r.x) && q.x >= Mathf.Min(p.x, r.x) &&
21                      q.y <= Mathf.Max(p.y, r.y) && q.y >= Mathf.Min(p.y, r.y);
22           }
23   
24           public static int Orientation(Vector2 p, Vector2 q, Vector2 r)
25           {
26               float val = (q.y - p.y) * (r.x - q.x) -
27                         (q.x - p.x) * (r.y - q.y);
28    
29               // collinear
30               if (val == 0) return 0; 
31    
32               // clock or counterclock wise
33               return val > 0? 1: 2; 
34           }
35           
36           public static bool IsIntersecting(LineSegment lhs, LineSegment rhs)
37           {
38               Vector2 p1 = lhs.A;
39               Vector2 q1 = lhs.B;
40               Vector2 p2 = rhs.A;
41               Vector2 q2 = rhs.B;
42               
43               // Find the four orientations needed for general and
44               // special cases
45               int o1 = Orientation(p1, q1, p2);
46               int o2 = Orientation(p1, q1, q2);
47               int o3 = Orientation(p2, q2, p1);
48               int o4 = Orientation(p2, q2, q1);
49    
50               // General case
51               if (o1 != o2 && o3 != o4)
52                   return true;
53    
54               // Special Cases
55               // p1, q1 and p2 are collinear and p2 lies on segment p1q1
56               if (o1 == 0 && ArePointsOnSegment(p1, p2, q1)) return true;
57    
58               // p1, q1 and q2 are collinear and q2 lies on segment p1q1
59               if (o2 == 0 && ArePointsOnSegment(p1, q2, q1)) return true;
60    
61               // p2, q2 and p1 are collinear and p1 lies on segment p2q2
62               if (o3 == 0 && ArePointsOnSegment(p2, p1, q2)) return true;
63    
64               // p2, q2 and q1 are collinear and q1 lies on segment p2q2
65               if (o4 == 0 && ArePointsOnSegment(p2, q1, q2)) return true;
66    
67               return false; // Doesn't fall in any of the above cases
68           }
69       }
70   }

1    using System.Collections.Generic;
2    using UnityEngine;
3    using Vector2 = UnityEngine.Vector2;
4    
5    namespace Shapes
6    {
7        public class AreaController : MonoBehaviour
8        {
9            [SerializeField]
10           protected List<Transform> shapeNodes;
11           public List<Transform> ShapeNodes => shapeNodes;
12   
13           [SerializeField]
14           protected List<Vector2> shapeVertices;
15           public List<Vector2> ShapeVertices { get; set; }
16   
17           [SerializeField]
18           protected List<LineSegment> shapeEdges;
19   
20           private void Awake()
21           {
22               // Store points
23               shapeVertices = new List<Vector2>();
24               foreach (var node in shapeNodes)
25               {
26                   shapeVertices.Add(node.position);
27               }
28               
29               // Generate Edges
30               var shapeVerticesCount = shapeVertices.Count;
31               for (var i = 0; i < shapeVerticesCount; i++)
32               {
33                   var currentVertex = shapeVertices[i];
34                   // Edge case: Last point loops back to initial point
35                   if (i == shapeVerticesCount - 1)
36                   {
37                       shapeEdges.Add(new LineSegment(currentVertex, shapeVertices[0]));
38                       continue;
39                   }
40                   var nextVertex = shapeVertices[i + 1];
41                   shapeEdges.Add(new LineSegment(currentVertex, nextVertex));
42               }
43           }
44   
45           public bool IsPointInShape(Vector2 point)
46           {
47               var lineSegment = new LineSegment(point, point + Vector2.right * float.MaxValue);
48               
49               // Loop through all edges
50               foreach (var edge in  shapeEdges)
51               {
52                   if (LineSegment.IsIntersecting(lineSegment, edge))
53                   {
54                       return true;
55                   }
56               }
57               return false;
58           }
59       }
60   }

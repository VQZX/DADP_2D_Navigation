1    using System.Collections.Generic;
2    using UnityEngine;
3    using UnityEngine.Serialization;

5    namespace Waypoint

7        public class WaypointController : MonoBehaviour
8        {
9            [FormerlySerializedAs("shapeNodes")]
10           [SerializeField]
11           protected List<Transform> waypointNodes;
12           public List<Transform> WaypointNodes => waypointNodes;
13   
14           public Transform GetClosestNode(Vector2 worldPoint)
15           {
16               float closestDistance = float.MaxValue;
17               Transform currentClosest = null;
18               foreach (var node in waypointNodes)
19               {
20                   var distance = Vector2.Distance(worldPoint, node.position);
21                   if (distance < closestDistance)
22                   {
23                       closestDistance = distance;
24                       currentClosest = node;
25                   }
26               }
27               return currentClosest;
28           }
29   
30           
31           // Allow going backward and forwards
32           public List<Transform> GetPath(Transform beginNode, Transform endNode)
33           {
34               if (!waypointNodes.Contains(beginNode))
35               {
36                   beginNode = waypointNodes[0];
37               }
38               var list = new List<Transform>();
39               var addToPath = false;
40   
41               var beginIndex = waypointNodes.IndexOf(beginNode);
42               var endIndex = waypointNodes.IndexOf(endNode);
43   
44               if (beginIndex < endIndex)
45               {
46                   for (var index = beginIndex; index <= endIndex; index++)
47                   {
48                       Debug.Log(index);
49                       var current = waypointNodes[index];
50                       list.Add(current);
51                   }
52               }
53               else if (endIndex < beginIndex)
54               {
55                   for (var index = beginIndex; index >= endIndex; index--)
56                   {
57                       Debug.Log(index);
58                       var current = waypointNodes[index];
59                       list.Add(current);
60                   }
61               }
62               return list;
63           }
64       }
